# Runtime dependencies for AlForks.
# Install: pip install -r requirements.txt

# Pinned to known-good versions. Bump deliberately when upgrading.

# Web framework — used in app.py
Flask>=3.0,<4.0

# GPX parsing — used in app.py (the dominant cost in parse_gpx;
# replacing it with xml.etree.ElementTree is on the enhancement backlog)
gpxpy>=1.6,<2.0

# Garmin Connect HR sync — used by sync/garmin_sync.py only.
# Optional for the core viewer; required if you run --login / --sync.
garminconnect>=0.3,<0.4

# IANA timezone database for zoneinfo. REQUIRED on Windows (no system tz db),
# which Python's zoneinfo needs to resolve names like "America/Edmonton". Without
# it, ZoneInfo() raises and _merge_hr_into_data silently drops HR — i.e. heart
# rate, HR zones, training load and weather all quietly disappear.
tzdata>=2024.1
